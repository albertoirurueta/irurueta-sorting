16  package com.irurueta.sorting;
17  
18  /**
19   * Interface for objects that can be compared and averaged with other objects.
20   *
21   * @param <T> Type to be compared and averaged
22   */
23  public interface ComparableAndAverageable<T> extends Comparable<T> {
24  
25      /**
26       * Averages current instance with another instance. (i.e. if both objects
27       * where Doubles, then the result would be equal to 0.5 * (obj1 + obj2)).
28       *
29       * @param other Other instance to be averaged with.
30       * @return An instance representing the average of both instances.
31       */
32      T averageWith(final T other);
33  }
